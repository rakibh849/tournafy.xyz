import os from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup from telegram.ext import Application, CommandHandler, CallbackQueryHandler, ContextTypes from dotenv import load_dotenv # .env ফাইল থেকে TOKEN নেবে load_dotenv() TOKEN = os.getenv("BOT_TOKEN") # /start কমান্ড async def start(update: Update, context: ContextTypes.DEFAULT_TYPE): user = update.effective_user keyboard = [ [InlineKeyboardButton("📝 কাজ শুরু করি", callback_data="tasks")], [InlineKeyboardButton("👥 রেফারাল লিঙ্ক", callback_data="referral")], [InlineKeyboardButton("💰 ব্যালেন্স", callback_data="balance")], ] await update.message.reply_text( f"👋 হ্যালো {user.first_name}!\nএখানে কাজ করে আয় করতে পারবেন।", reply_markup=InlineKeyboardMarkup(keyboard) ) # বাটন হ্যান্ডলার async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE): query = update.callback_query await query.answer() if query.data == "tasks": await query.edit_message_text("📋 এখানে কাজগুলা আসবে...") elif query.data == "referral": ref_link = f"https://t.me/{context.bot.username}?start={query.from_user.id}" await query.edit_message_text(f"🔗 আপনার রেফারাল লিঙ্ক:\n{ref_link}") elif query.data == "balance": # ডাটাবেস এ ইউজারের কয়েন/টাকা রাখতে হবে (এখন শুধু demo) await query.edit_message_text("💰 আপনার ব্যালেন্স: 0 টাকা") # মেইন রানার def main(): app = Application.builder().token(TOKEN).build() app.add_handler(CommandHandler("start", start)) app.add_handler(CallbackQueryHandler(button_handler)) app.run_polling() if __name__ == "__main__": main()
